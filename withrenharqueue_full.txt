221   const prev = renharOpQueueRef.current[key] || Promise.resolve();
222   const thisOp = prev.then(async () => {
223     const { data } = await supabase.from("renhar").select("*")
224       .eq("raw_id", task.rawId).eq("wp", task.wp).eq("tanggal", task.tanggal)
225       .order("updated_at", { ascending: false, nullsFirst: false }).limit(1);
226     await fn(data?.[0] || null);
227   });
228   renharOpQueueRef.current[key] = thisOp;
229   await thisOp;
